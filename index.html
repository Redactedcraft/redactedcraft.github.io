Our custom JSON-based modeling system allows for complex block shapes like doors with protruding 3D knobs.

// 3D Door Model Definition
private static BlockModel CreateDoorModel(bool open) {
    var elements = new List<BlockModelElement>();
    // Main Door Slab
    elements.Add(new BlockModelElement {
        From = Vector3.Zero,
        To = open ? new Vector3(2, 32, 16) : new Vector3(16, 32, 2)
    });
    // 3D Knob
    elements.Add(new BlockModelElement {
        From = open ? new Vector3(2, 12, 12) : new Vector3(12, 12, 2),
        To = open ? new Vector3(4, 14, 14) : new Vector3(14, 14, 4)
    });
    return new BlockModel(elements, display, new Vector2(48, 32));
}
